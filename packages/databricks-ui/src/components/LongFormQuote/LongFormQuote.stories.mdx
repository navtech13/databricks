# LongFormQuote

LongFormQuote component hosts a title and a description.

## Props

<Props of={LongFormQuote} />

<Preview>
  <Story name='Long Form Quote'>
    {(args) => {
      const [longFormQuote, setLongFormQuote] = useState(null)
      useEffect(() => {
        fetch(`/longFormQuote`)
          .then((response) => response.json())
          .then(({ data }) => setLongFormQuote(data))
      }, [])
      if (!longFormQuote) {
        return <></>
      }
      return (
        <LongFormQuote
          source={args.source || longFormQuote.source}
        >
          {args.children || longFormQuote.text}
        </LongFormQuote>
      )
    }}
  </Story>
</Preview>
