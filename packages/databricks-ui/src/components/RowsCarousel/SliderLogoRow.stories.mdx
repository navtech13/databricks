# RowsCarousel

Component for displaying Rows as a Carousel on small viewports. The rest of the props will be applied to the Carousel component (like loop and autoplay).

<Props of={RowsCarousel} />

## 3.2 Slider Logo Row

<Preview>
  <Story name='Slider Logo Row'>
    {(args) => {
      const [rowsCarousel, setRowsCarousel] = useState(null)
      useEffect(() => {
        fetch(`/rowsCarousel`)
          .then((response) => response.json())
          .then(({ data }) => setRowsCarousel(data))
      }, [])
      if (!rowsCarousel) {
        return <></>
      }
      return (
        <RowsCarousel
          multipleRows={false}
          columns={4}
          gap={3}
          footer='+ Any other Apache Spark™️ compatible client'
        >
          {rowsCarousel.logos.slice(0, 6).map(({ src, alt }) => {
            return (
              <Image
                key={src}
                className={"h-full"}
                imageOptions={{
                  className:
                    "w-full h-full object-contain object-center-center mx-auto ",
                }}
                src={src}
                alt={alt}
              />
            )
          })}
        </RowsCarousel>
      )
    }}
  </Story>
</Preview>
