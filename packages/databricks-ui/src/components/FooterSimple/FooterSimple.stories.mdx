### FooterSimple

<Preview>
  <Story name='Footer Simple' args={{ items: 3 }}>
    {(args) => {
      const [footerSimple, setFooterSimple] = useState(null)
      useEffect(() => {
        fetch(`/footerSimple`)
          .then((response) => response.json())
          .then(({ data }) => setFooterSimple(data))
      }, [])
      if (!footerSimple) {
        return <></>
      }
      return <FooterSimple items={footerSimple.slice(0, args.items)} />
    }}
  </Story>
