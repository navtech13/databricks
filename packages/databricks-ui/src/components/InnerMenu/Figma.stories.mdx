<Preview>
  <Story name='Link List' args={{ items: 8 }}>
    {(args) => {
      const [innerMenu, setInnerMenu] = useState(null)
      useEffect(() => {
        fetch(`/innerMenu`)
          .then((response) => response.json())
          .then(({ data }) => setInnerMenu(data))
      }, [])
      if (!innerMenu) {
        return <></>
      }
      return <InnerMenu items={innerMenu.slice(0, args.items)} />
    }}
  </Story>
