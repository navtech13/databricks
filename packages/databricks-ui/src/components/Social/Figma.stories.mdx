<Preview>
  <Story name='Longform Social Sharing'>
    {(args) => {
      const [social, setSocial] = useState(null)
      useEffect(() => {
        fetch(`/social`)
          .then((response) => response.json())
          .then(({ data }) => setSocial(data))
      }, [])
      if (!social) {
        return <></>
      }
      return (
        <div style={{ padding: "20px" }}>
          <Social items={social.items}>{args.children || social.children}</Social>
        </div>
      )
    }}
