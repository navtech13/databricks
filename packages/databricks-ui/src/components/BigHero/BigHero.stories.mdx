# BigHero

Hero Component, showing an animation that can be paused.

## Props

<Props of={BigHero} />

### BigHero

<Preview>
  <Story name='Homepage Hero'>
    {(args) => {
      const [bigHero, setBigHero] = useState(null)
      useEffect(() => {
        fetch(`/bigHero`)
          .then((response) => response.json())
          .then(({ data }) => setBigHero(data))
      }, [])
      if (!bigHero) {
        return <></>
      }
      return (
        <BigHero
          typeWriterMessages={bigHero.typeWriterMessages}
          ctas={bigHero.ctas}
          typeWriterTopText={args.typeWriterTopText || bigHero.typeWriterTopText}
          typeWriterBottomText={
            args.typeWriterBottomText || bigHero.typeWriterBottomText
          }
          lottie={bigHero.lottie}
          fallbackImage={bigHero.fallbackImage}
          description={args.description || bigHero.description}
          typeWriterPauseTime={bigHero.typeWriterPauseTime}
          typeWriterDelayTime={bigHero.typeWriterDelayTime}
          backgroundImage={bigHero.backgroundImage}
        />
      )
    }}
  </Story>
</Preview>
