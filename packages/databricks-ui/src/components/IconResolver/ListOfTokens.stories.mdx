### Token List

<Preview>
  <Story name='List of Tokens '>
    {() => {
      const [icons, setIcon] = useState(null)
      useEffect(() => {
        fetch(`/icons`)
          .then((response) => response.json())
          .then(({ data }) => setIcon(data))
      }, [])
      if (!icons) {
        return <></>
      }
      return (
        <div className='p-4'>
          {icons.map((icon) => {
            return (
              <div className=' m-2.5 flex flex-col items-center justify-center'>
                <IconResolver className='mb-2.5' token={icon} />
                <span>{icon}</span>
              </div>
            )
          })}
        </div>
      )
    }}
  </Story>
</Preview>
