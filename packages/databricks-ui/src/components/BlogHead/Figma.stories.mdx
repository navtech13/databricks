### 2.3 Longform Header

<Preview>
  <Story
    name='Longform Header'
    args={{
      subtitle:
        "Enim tempus amet, ac cras sit ut malesuada. Pellentesque duis etiam.",
    }}
  >
    {(args) => {
      const [blogHead, setBlogHead] = useState(null)
      useEffect(() => {
        fetch(`/blogHead`)
          .then((response) => response.json())
          .then(({ data }) => setBlogHead(data))
      }, [])
      if (!blogHead) {
        return <></>
      }
      return (
        <BlogHead
          className='mx-auto lg:w-8/12'
          title={args.title || blogHead.title}
          subtitle={args.subtitle}
        />
      )
    }}
  </Story>
