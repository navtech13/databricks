<Preview>
  <Story name='Cta Row Right' args={{ count: 3, position: "right" }}>
    {(args) => {
      const [ctaRow, setCtaRow] = useState(null)
      useEffect(() => {
        fetch(`/ctaRow`)
          .then((response) => response.json())
          .then(({ data }) => setCtaRow(data))
      }, [])
      if (!ctaRow) {
        return <></>
      }
      return (
        <div>
          <CtaRow
            items={ctaRow.item.slice(0, args.count)}
            position={args.position}
          />
        </div>
      )
    }}
  </Story>
