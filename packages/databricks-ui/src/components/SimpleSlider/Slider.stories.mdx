  <Story name='Simple Slider'>
    {(args) => {
      const [slider, setSlider] = useState(null)
      useEffect(() => {
        fetch(`/teamSlider`)
          .then((response) => response.json())
          .then(({ data }) => setSlider(data))
      }, [])
      if (!slider) {
        return <></>
      }
      return (
        <div style={{ margin: "20px" }}>
          <Slider items={sliderData} />
        </div>
      )
    }}
